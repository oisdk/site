\begin{code}

import Agda.Builtin.Nat as Builtin

conv : Builtin.Nat → ℕ
conv Builtin.zero = zero
conv (Builtin.suc n) = suc (conv n)

instance
  numberNat : Number ℕ
  Number.Constraint numberNat = λ _ → ⊤
  Number.fromNat numberNat n = conv n

\end{code}
