  diff : (x y : 𝑀) → ∃ (unsign (λ k → (y ≡ x ∙ k)) (x ≡ y) (λ k → (x ≡ y ∙ k)))
  diff x y with x ≤? y
  diff x y | inl (k , x≤y) = ⁻ k , x≤y
  diff x y | inr (k , x≥y) = ⁺ k , x≥y

\begin{code}
  ∙-cong : ∀ x {y z} → y ≤ z → x ∙ y ≤ x ∙ z
  ∙-cong x (k , z≡y∙k) = k , cong (x ∙_) z≡y∙k ; sym (assoc x _ k)

  ∙-congʳ : ∀ x {y z} → y ≤ z → y ∙ x ≤ z ∙ x
  ∙-congʳ x {y} {z} p = subst (y ∙ x ≤_) (comm x z) (subst (_≤ x ∙ z) (comm x y) (∙-cong x p))

\begin{code}

  monPartialOrder : PartialOrder 𝑀 ℓ
  PartialOrder._≤_ monPartialOrder = _≤_
  PartialOrder.refl monPartialOrder = ≤-refl
  PartialOrder.antisym monPartialOrder = antisym
  PartialOrder.trans monPartialOrder = ≤-trans

  totalOrder : TotalOrder 𝑀 ℓ ℓ
  totalOrder = fromPartialOrder monPartialOrder _≤?_

  open TotalOrder totalOrder hiding (refl; antisym; _≤_; _≤?_; ≤-trans) public

  ∙-distrib-⊓ : _∙_ Distributesˡ _⊓_
  ∙-distrib-⊓ x y z with x <? y | (x ∙ z) <? (y ∙ z)
  ∙-distrib-⊓ x y z | yes x₁  | yes x₂ = refl
  ∙-distrib-⊓ x y z | yes x₁  | no x₂  = antisym (∙-congʳ z (<⇒≤ x₁)) (≮⇒≥ x₂)
  ∙-distrib-⊓ x y z | no  x≥y | yes x∙z≤y∙z = antisym (∙-congʳ z (≮⇒≥ x≥y)) (<⇒≤ x∙z≤y∙z)
  ∙-distrib-⊓ x y z | no  x₁  | no x₂ = refl

  module MinPlus (sup : Sup) where
    Ω : 𝑀
    Ω = fst sup

    maximal : ∀ x → x ≤ Ω
    maximal = snd sup

    max-sup : ∀ x → Ω ⊓ x ≡ x
    max-sup x with Ω <? x
    max-sup x | yes x₁ = ⊥-elim (x₁ (maximal x))
    max-sup x | no  x₁ = refl

    cmb-sup : ∀ x → Ω ∙ x ≡ Ω
    cmb-sup x = antisym (maximal (Ω ∙ x)) (x , refl)

    max-plus : Semiring ℓ
    NearSemiring.𝑅 (Semiring.nearSemiring max-plus) = 𝑀
    NearSemiring._+_ (Semiring.nearSemiring max-plus) = _⊓_
    NearSemiring._∗_ (Semiring.nearSemiring max-plus) = _∙_
    NearSemiring.1# (Semiring.nearSemiring max-plus) = ε
    NearSemiring.0# (Semiring.nearSemiring max-plus) = Ω
    NearSemiring.+-assoc (Semiring.nearSemiring max-plus) = ⊓-assoc
    NearSemiring.∗-assoc (Semiring.nearSemiring max-plus) = assoc
    NearSemiring.0+ (Semiring.nearSemiring max-plus) = max-sup
    NearSemiring.+0 (Semiring.nearSemiring max-plus) x = ⊓-comm x Ω ; max-sup x
    NearSemiring.1∗ (Semiring.nearSemiring max-plus) = ε∙
    NearSemiring.∗1 (Semiring.nearSemiring max-plus) = ∙ε
    NearSemiring.0∗ (Semiring.nearSemiring max-plus) = cmb-sup
    NearSemiring.⟨+⟩∗ (Semiring.nearSemiring max-plus) = ∙-distrib-⊓
    Semiring.+-comm max-plus = ⊓-comm
    Semiring.∗0 max-plus x = comm x Ω ; cmb-sup x
    Semiring.∗⟨+⟩ max-plus x y z = comm x (y ⊓ z) ; ∙-distrib-⊓ y z x ; cong₂ _⊓_ (comm y x) (comm z x)
