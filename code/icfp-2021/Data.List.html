\begin{code}
{-# OPTIONS --without-K --safe #-}

module Data.List where

open import Level

-- We can also import the list definition from
-- Agda.Builtin.List if we need interop

data List (A : Type a) : Type a where
  [] : List A
  _∷_ : A → List A → List A


map : (A → B) → List A → List B
map f [] = []
map f (x ∷ xs) = f x ∷ map f xs

module _ {p : Level} where
\end{code}
%<*list-elim>
\begin{code}
  list-elim :  (P : List A → Type p) →
               (∀ x xs → P xs → P (x ∷ xs)) →
               P [] →
               ∀ xs → P xs
  list-elim P f b []        = b
  list-elim P f b (x ∷ xs)  = f x xs (list-elim P f b xs)
\end{code}
%</list-elim>
\begin{code}
foldr : (A → B → B) → B → List A → B
foldr f b [] = b
foldr f b (x ∷ xs) = f x (foldr f b xs)
\end{code}
