{-# OPTIONS --cubical --safe #-}

module Algebra.Construct.Free.Semiring.Eliminators where

open import Level
open import HLevels
open import Algebra
open import Path
open import Algebra.Construct.Free.Semiring.Definition

record Eliminator (A : Type a) {p} (P : A * → Type p) : Type (p ℓ⊔ a) where
  constructor elim
  field
    ⟦_⟧-set : ∀ {xs} → isSet (P xs)
    ⟦_⟧_⊗_⟨_⊗_⟩ : ∀ x y → P x → P y → P (x ⊗ y)
    ⟦_⟧_⊕_⟨_⊕_⟩ : ∀ x y → P x → P y → P (x ⊕ y)
    ⟦_⟧⓪        : P ⓪
    ⟦_⟧①        : P ①
  private t = ⟦_⟧_⊗_⟨_⊗_⟩; s = ⟦_⟧_⊕_⟨_⊕_⟩; z = ⟦_⟧⓪; o = ⟦_⟧①
  field
    ⟦_⟧-⊕-comm : ∀ x y px py → s x y px py ≡[ i ≔ P (⊕-comm x y i) ]≡ s y x py px
    ⟦_⟧-⊗-comm : ∀ x y px py → t x y px py ≡[ i ≔ P (⊗-comm x y i) ]≡ t y x py px

    ⟦_⟧-⊕-assoc : ∀ x y z px py pz → s (x ⊕ y) z (s x y px py) pz ≡[ i ≔ P (⊕-assoc x y z i) ]≡ s x (y ⊕ z) px (s y z py pz)
    ⟦_⟧-⊗-assoc : ∀ x y z px py pz → t (x ⊗ y) z (t x y px py) pz ≡[ i ≔ P (⊗-assoc x y z i) ]≡ t x (y ⊗ z) px (t y z py pz)

    ⟦_⟧⓪⊕ : ∀ x px → s ⓪ x z px ≡[ i ≔ P (⓪⊕ x i) ]≡ px
    ⟦_⟧①⊗ : ∀ x px → t ① x o px ≡[ i ≔ P (①⊗ x i) ]≡ px

    ⟦_⟧⓪⊗ : ∀ x px → t ⓪ x z px ≡[ i ≔ P (⓪⊗ x i) ]≡ z

    ⟦_⟧⊗⊕ : ∀ x y z px py pz → t x (y ⊕ z) px (s y z py pz) ≡[ i ≔ P (⊗⊕ x y z i) ]≡ s (x ⊗ y) (x ⊗ z) (t x y px py) (t x z px pz)

  run : (xs : A *) → P xs
  run (xs ⊗ ys) = t xs ys (run xs) (run ys)
  run (xs ⊕ ys) = s xs ys (run xs) (run ys)
  run ① = o
  run ⓪ = z

  run (⊕-assoc xs ys zs i) = ⟦_⟧-⊕-assoc xs ys zs (run xs) (run ys) (run zs) i
  run (⊗-assoc xs ys zs i) = ⟦_⟧-⊗-assoc xs ys zs (run xs) (run ys) (run zs) i
  run (⊕-comm xs ys i) = ⟦_⟧-⊕-comm xs ys (run xs) (run ys) i
  run (⊗-comm xs ys i) = ⟦_⟧-⊗-comm xs ys (run xs) (run ys) i
  run (⓪⊕ xs i) = ⟦_⟧⓪⊕ xs (run xs) i
  run (①⊗ xs i) = ⟦_⟧①⊗ xs (run xs) i
  run (⓪⊗ xs i) = ⟦_⟧⓪⊗ xs (run xs) i
  run (⊗⊕ xs ys zs i) = ⟦_⟧⊗⊕ xs ys zs (run xs) (run ys) (run zs) i
  run (trunc xs ys p q i j) =
      isOfHLevel→isOfHLevelDep 2
        (λ xs → ⟦_⟧-set {xs})
        (run xs) (run ys)
        (cong run p) (cong run q)
        (trunc xs ys p q)
        i j

  _⇓_ : (xs : A *) → P xs
  _⇓_ = run
  {-# INLINE _⇓_ #-}

open Eliminator public
infixr 1 Eliminator

syntax Eliminator A (λ v → e) = ∗-Π[ v ⦂ e ] A

\end{code}
