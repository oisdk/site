data _* (A : Type a) : Type a where
  _⊗_ : A * → A * → A *
  _⊕_ : A * → A * → A *
  ① : A *
  ⓪ : A *

  ⊕-assoc : Associative _⊕_
  ⊗-assoc : Associative _⊗_

  ⊕-comm : Commutative _⊕_
  ⊗-comm : Commutative _⊗_

  ⓪⊕ : Identityˡ _⊕_ ⓪
  ①⊗ : Identityˡ _⊗_ ①

  ⓪⊗ : ⓪ Annihilatesˡ _⊗_

  ⊗⊕ : _⊗_ Distributesʳ _⊕_

  trunc : isSet (A *)
