{-# OPTIONS --cubical --safe #-}

open import Algebra
open import Relation.Binary

module Algebra.Monus.HaskellPerspective
  {ℓ₁ ℓ₂ ℓ₃}
  (mon : CommutativeMonoid ℓ₁)
  (rel : TotalOrder (CommutativeMonoid.𝑆 mon) ℓ₂ ℓ₃)
  (_∸_ : CommutativeMonoid.𝑆 mon → CommutativeMonoid.𝑆 mon → CommutativeMonoid.𝑆 mon)
  where

open CommutativeMonoid mon
open TotalOrder rel
open import Path
open import Data.Sum
open import Data.Sigma

module _ (law₁ : ∀ x y → x ≤ x ∙ y) (law₂ : ∀ x y → x ≤ y → x ∙ (x ∸ y) ≡ y) where

  open import Algebra.Monus

  isMonus : Monus _
  Monus.commutativeMonoid isMonus = mon
  Monus._≤?_ isMonus x y = map-⊎ (λ x≤y → x ∸ y , sym (law₂ x y x≤y)) (λ y≤x → y ∸ x , sym (law₂ y x y≤x)) (x ≤|≥ y)
  Monus.antisym isMonus {x} {y} (k₁ , x≤y) (k₂ , y≤x) =
    antisym (subst (x ≤_) (sym x≤y) (law₁ x k₁)) (subst (y ≤_) (sym y≤x) (law₁ y k₂))
