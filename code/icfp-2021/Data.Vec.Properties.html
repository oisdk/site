{-# OPTIONS --cubical --safe #-}

module Data.Vec.Properties where

open import Path
open import Level
open import Data.Nat
open import Function
open import Data.Fin
open import Data.Vec

private
  variable
    n m : ℕ

lookup-replace : (i : Fin n) (x : A) (xs : Vec A n) → lookup i (replace i x xs) ≡ x
lookup-replace f0     x xs = refl
lookup-replace (fs i) x (_ ∷ xs) = lookup-replace i x xs

replace-replace : (i : Fin n) (x y : A) (xs : Vec A n) → replace i x (replace i y xs) ≡ replace i x xs
replace-replace f0     x y xs       = refl
replace-replace (fs i) x y (z ∷ xs) = cong (z ∷_) (replace-replace i x y xs)
