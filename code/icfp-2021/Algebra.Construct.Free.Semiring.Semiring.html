{-# OPTIONS --cubical --safe --postfix-projections #-}

module Algebra.Construct.Free.Semiring.Semiring where

open import Level
open import HLevels
open import Algebra
open import Path
open import Algebra.Construct.Free.Semiring.Definition

module _ {a} {A : Type a} where
  open CommutativeSemiring
  open Semiring
  open NearSemiring

  freeSemiringSemiring : CommutativeSemiring a
  freeSemiringSemiring .semiring .nearSemiring .𝑅          = A *
  freeSemiringSemiring .semiring .nearSemiring ._+_        = _⊕_
  freeSemiringSemiring .semiring .nearSemiring ._∗_        = _⊗_
  freeSemiringSemiring .semiring .nearSemiring .1#         = ①
  freeSemiringSemiring .semiring .nearSemiring .0#         = ⓪
  freeSemiringSemiring .semiring .nearSemiring .+-assoc    = ⊕-assoc
  freeSemiringSemiring .semiring .nearSemiring .∗-assoc    = ⊗-assoc
  freeSemiringSemiring .semiring .nearSemiring .0+         = ⓪⊕
  freeSemiringSemiring .semiring .nearSemiring .+0 x       = ⊕-comm x ⓪ ; ⓪⊕ x
  freeSemiringSemiring .semiring .nearSemiring .1∗         = ①⊗
  freeSemiringSemiring .semiring .nearSemiring .∗1 x       = ⊗-comm x ① ; ①⊗ x
  freeSemiringSemiring .semiring .nearSemiring .0∗         = ⓪⊗
  freeSemiringSemiring .semiring .nearSemiring .⟨+⟩∗ x y z = ⊗-comm (x ⊕ y) z ; ⊗⊕ z x y ; cong₂ _⊕_ (⊗-comm z x) (⊗-comm z y)
  freeSemiringSemiring .semiring .+-comm                   = ⊕-comm
  freeSemiringSemiring .semiring .∗0 x                     = ⊗-comm x ⓪ ; ⓪⊗ x
  freeSemiringSemiring .semiring .∗⟨+⟩                     = ⊗⊕
  freeSemiringSemiring .∗-comm                             = ⊗-comm
\end{code}
