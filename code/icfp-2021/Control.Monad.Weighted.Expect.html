%<*expect-alg>
\begin{code}
∫ : (A → 𝑅) → W-ϕ[ A ] 𝑅
[ ∫ f ] p & x ∷ _ ⟨ ∫fxs ⟩   = p ∗ f x + ∫fxs
[ ∫ f ][]                    = 0#
\end{code}
%</expect-alg>
%<*com>
\begin{code}
[ ∫ f ]-com p x q y _ r =
  p ∗ f x + (q ∗ f y + r)
    ≡˘⟨ +-assoc _ _ r ⟩
  p ∗ f x + q ∗ f y + r
    ≡⟨ cong (_+ r) (+-com _ _) ⟩
  q ∗ f y + p ∗ f x + r
    ≡⟨ +-assoc _ _ r ⟩
  q ∗ f y + (p ∗ f x + r) ∎
\end{code}
%</com>
%<*dup>
\begin{code}
[ ∫ f ]-dup p q x _ r =
  p ∗ f x + (q ∗ f x + r)
    ≡˘⟨ +-assoc _ _ r ⟩
  (p ∗ f x + q ∗ f x) + r
    ≡˘⟨ cong (_+ r) (⟨+⟩∗ _ _ _) ⟩
  (p + q) ∗ f x + r ∎
\end{code}
%</dup>
%<*del>
\begin{code}
[ ∫ f ]-del x _ r =
  0# ∗ f x + r
    ≡⟨ cong (_+ r) (0∗ (f x)) ⟩
  0# + r
    ≡⟨ 0+ r ⟩
  r ∎
\end{code}
%</del>
