fromMaybe : A → Maybe A → A
fromMaybe x nothing = x
fromMaybe _ (just y) = y

discreteMaybe : Discrete A → Discrete (Maybe A)
discreteMaybe eq nothing nothing = yes refl
discreteMaybe eq nothing (just x) = no (λ p → subst (maybe ⊤ (const ⊥)) p tt)
discreteMaybe eq (just x) nothing = no (λ p → subst (maybe ⊥ (const ⊤)) p tt)
discreteMaybe eq (just x) (just y) = ⟦yes cong just ,no cong (fromMaybe x) ⟧ (eq x y)
