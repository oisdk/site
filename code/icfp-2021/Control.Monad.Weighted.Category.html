%<*functor>
\begin{code}
data W-F (A : Type a) (B : Type b) : Type (ℓ ℓ⊔ a ℓ⊔ b) where
  []       : W-F A B
  _&_∷_⟨_⟩ : 𝑅 → A → W A → B → W-F A B
\end{code}
%</functor>
%<*eval>
\begin{code}
_↓↓_ : W-ϕ[ A ] B → W-F A B → B
a ↓↓ [] = [ a ][]
a ↓↓ (p & x ∷ xs ⟨ pxs ⟩) = [ a ] p & x ∷ xs ⟨ pxs ⟩
\end{code}
%</eval>
%<*map>
\begin{code}
map : (B → C) → W-F A B → W-F A C
map f [] = []
map f (x & p ∷ xs ⟨ pxs ⟩) = x & p ∷ xs ⟨ f pxs ⟩
\end{code}
%</map>
\begin{code}
module _ {v} {V : Type v} {𝒸 : Level} where

  W-Alg : Type _
\end{code}
%<*w-alg>
\begin{code}
  W-Alg = Σ[ A ⦂ Type 𝒸 ] (W-ϕ[ V ] A)
\end{code}
%</w-alg>
\begin{code}
  _⟶_ : W-Alg → W-Alg → Type _
\end{code}
%<*w-alg-hom>
\begin{code}
  (A , a) ⟶ (B , b) = Σ[ h ⦂ (A → B) ] (∀ x → h (a ↓↓ x) ≡ b ↓↓ map h x)
\end{code}
%</w-alg-hom>
\begin{code}

  private variable X Y Z : W-Alg

  id′ : X ⟶ X
  id′ .fst = id
  id′ .snd [] = refl
  id′ .snd (_ & _ ∷ _ ⟨ _ ⟩) = refl

  _∘′_ : (Y ⟶ Z) → (X ⟶ Y) → (X ⟶ Z)
  (f ∘′ g) .fst = f .fst ∘ g .fst
  (f ∘′ g) .snd [] = cong (f .fst) (g .snd []) ; f .snd []
  (f ∘′ g) .snd (p & x ∷ xs ⟨ pxs ⟩) = cong (f .fst) (g .snd (p & x ∷ xs ⟨ pxs ⟩)) ; f .snd (p & x ∷ xs ⟨ g .fst pxs ⟩ )
\end{code}
