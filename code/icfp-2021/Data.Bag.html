data ⟅_⟆ (A : Type a) : Type a where
  []     : ⟅ A ⟆
  _∷_    : A → ⟅ A ⟆ → ⟅ A ⟆
  com    : ∀ x y xs → x ∷ y ∷ xs ≡ y ∷ x ∷ xs
  trunc  : isSet ⟅ A ⟆

\begin{code}

module Eliminate
  {a p}
  (A : Type a)
  (P : ⟅ A ⟆ → Type p)
  (isSetP :  ∀ {xs} → isSet (P xs))
  (nil : P [])
  (cons : ∀ x xs → P xs → P (x ∷ xs))
  (ccom : ∀ x y xs pxs →
      cons x (y ∷ xs) (cons y xs pxs) ≡[ i ≔ P (com x y xs i) ]≡
      cons y (x ∷ xs) (cons x xs pxs))
  where
  eliminate : ∀ xs → P xs
  eliminate [] = nil
  eliminate (x ∷ xs) = cons x xs (eliminate xs)
  eliminate (com x y xs i) = ccom x y xs (eliminate xs)  i
  eliminate (trunc xs ys x y i j) =
      isOfHLevel→isOfHLevelDep 2
        (λ xs → isSetP {xs})
        (eliminate xs) (eliminate ys)
        (cong eliminate x) (cong eliminate y)
        (trunc xs ys x y)
        i j

record Elim {a p} (A : Type a) (P : ⟅ A ⟆ → Type p) : Type (a ℓ⊔ p) where
  no-eta-equality
  constructor elim
  field
    ⟦_⟧-set : ∀ {xs} → isSet (P xs)
    ⟦_⟧[] : P []
    ⟦_⟧_∷_⟨_⟩ : ∀ x xs → P xs → P (x ∷ xs)
  private z = ⟦_⟧[]; f = ⟦_⟧_∷_⟨_⟩
  field
    ⟦_⟧-com : ∀ x y xs pxs →
      f x (y ∷ xs) (f y xs pxs) ≡[ i ≔ P (com x y xs i) ]≡
      f y (x ∷ xs) (f x xs pxs)
  ⟦_⟧⇓ : ∀ xs → P xs
  ⟦_⟧⇓ = Eliminate.eliminate A P ⟦_⟧-set ⟦_⟧[] ⟦_⟧_∷_⟨_⟩ ⟦_⟧-com
  {-# INLINE ⟦_⟧⇓ #-}
open Elim

record Elim-prop {a p} (A : Type a) (P : ⟅ A ⟆  → Type p) : Type (a ℓ⊔ p) where
  no-eta-equality
  constructor elim-prop
  field
    ∥_∥-prop : ∀ {xs} → isProp (P xs)
    ∥_∥[] : P []
    ∥_∥_∷_⟨_⟩ : ∀ x xs → P xs → P (x ∷ xs)
  private z = ∥_∥[]; f = ∥_∥_∷_⟨_⟩
  ∥_∥⇓ : ∀ xs → P xs
  ∥_∥⇓ = Eliminate.eliminate A P
          (λ {xs} → isProp→isSet (∥_∥-prop {xs}))
          z f
          (λ x y xs pxs → toPathP (∥_∥-prop (transp (λ i → P (com x y xs i)) i0 (f x (y ∷ xs) (f y xs pxs))) (f y (x ∷ xs) (f x xs pxs))))
  {-# INLINE ∥_∥⇓ #-}
open Elim-prop

record Rec {a b} (A : Type a) (B : Type b) : Type (a ℓ⊔ b) where
  no-eta-equality
  constructor rec
  field
    [_]-set  : isSet B
    [_]_∷_   : A → B → B
    [_][]    : B
  private f = [_]_∷_; z = [_][]
  field
    [_]-com : ∀ x y xs → f x (f y xs) ≡ f y (f x xs)
  [_]↓ : ⟅ A ⟆ → B
  [_]↓ = Eliminate.eliminate A (λ _ → B)
            [_]-set
            z
            (λ x _ xs → f x xs)
            (λ x y xs → [_]-com x y)
  {-# INLINE [_]↓ #-}
open Rec

un-alg : ⟅ A ⟆ → Rec A ⟅ A ⟆
[ un-alg ys ]-set = trunc
[ un-alg ys ] x ∷ xs = x ∷ xs
[ un-alg ys ][] = ys
[ un-alg ys ]-com = com

infixr 5 _∪_
_∪_ : ⟅ A ⟆ → ⟅ A ⟆ → ⟅ A ⟆
xs ∪ ys = [ un-alg ys ]↓ xs

∪-assoc : (xs ys zs : ⟅ A ⟆) → (xs ∪ ys) ∪ zs ≡ xs ∪ (ys ∪ zs)
∪-assoc xs ys zs = ∥ assoc-prf ys zs ∥⇓ xs
  where
  assoc-prf : (ys zs : ⟅ A ⟆) → Elim-prop A (λ xs → (xs ∪ ys) ∪ zs ≡ xs ∪ (ys ∪ zs))
  ∥ assoc-prf ys zs ∥-prop = trunc _ _
  ∥ assoc-prf ys zs ∥[] = refl
  ∥ assoc-prf ys zs ∥ x ∷ xs ⟨ Pxs ⟩ = cong (x ∷_) Pxs

∪-cons : ∀ x (xs ys : ⟅ A ⟆) → x ∷ xs ∪ ys ≡ xs ∪ x ∷ ys
∪-cons x xs ys = ∥ cons-prf x ys ∥⇓ xs
  where
  cons-prf : ∀ x (ys : ⟅ A ⟆) → Elim-prop A (λ xs → x ∷ xs ∪ ys ≡ xs ∪ x ∷ ys)
  ∥ cons-prf x ys ∥-prop = trunc _ _
  ∥ cons-prf x ys ∥[] = refl
  ∥ cons-prf x ys ∥ y ∷ xs ⟨ Pxs ⟩ = com x y (xs ∪ ys) ; cong (y ∷_) Pxs

∪-idʳ : (xs : ⟅ A ⟆) → xs ∪ [] ≡ xs
∪-idʳ = ∥ id-prf ∥⇓
  where
  id-prf : Elim-prop A (λ xs → xs ∪ [] ≡ xs)
  ∥ id-prf ∥-prop = trunc _ _
  ∥ id-prf ∥[] = refl
  ∥ id-prf ∥ x ∷ xs ⟨ Pxs ⟩ = cong (x ∷_) Pxs

∪-com : (xs ys : ⟅ A ⟆) → xs ∪ ys ≡ ys ∪ xs
∪-com xs ys = ∥ com-prf ys ∥⇓ xs
  where
  com-prf : (ys : ⟅ A ⟆) → Elim-prop A (λ xs → xs ∪ ys ≡ ys ∪ xs)
  ∥ com-prf ys ∥-prop = trunc _ _
  ∥ com-prf ys ∥[] = sym (∪-idʳ ys)
  ∥ com-prf ys ∥ x ∷ xs ⟨ Pxs ⟩ = cong (x ∷_) Pxs ; ∪-cons x ys xs
