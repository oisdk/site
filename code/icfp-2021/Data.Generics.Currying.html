Curriedⁿ : ℕ → Type₁
Curriedⁿ zero    = Type
Curriedⁿ (suc n) = Type → Curriedⁿ n

_~_ : ∀ {n} → (Params n → Type) → Curriedⁿ n
_~_ {n = zero}  f = f []
_~_ {n = suc n} f A = _~_ {n = n} (f ∘ (A ∷_))
