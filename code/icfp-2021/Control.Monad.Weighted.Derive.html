min : S-ϕ[ A ] (Maybe 𝑀)
[ min ] w & x ∷ xs ⟨ nothing  ⟩ = just w
[ min ] w & x ∷ xs ⟨ just m   ⟩ = just (w ⊓ m)

%<*min-proofs>
\begin{code}
[ min ]-dup p q _ _ nothing = refl
[ min ]-dup p q _ _ (just r) = cong just (
  p ⊓ (q ⊓ r) ≡˘⟨ ⊓-assoc p q r ⟩
  (p ⊓ q) ⊓ r ∎)
[ min ]-com p _ q _ _ nothing = cong just (
  p ⊓ q ≡⟨ ⊓-comm p q ⟩
  q ⊓ p ∎)
[ min ]-com p _ q _ _ (just r) = cong just (
  p ⊓ (q ⊓ r)  ≡˘⟨ ⊓-assoc p q r ⟩
  (p ⊓ q) ⊓ r  ≡⟨ cong (_⊓ r) (⊓-comm p q) ⟩
  (q ⊓ p) ⊓ r  ≡⟨ ⊓-assoc q p r ⟩
  q ⊓ (p ⊓ r) ∎)
\end{code}
