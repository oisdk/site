\begin{code}
[∙]≢sup : ∀ {x : A} → [ x ] ≢ sup
[∙]≢sup [x]≡sup = subst (bool ⊤ ⊥) (cong isSup [x]≡sup) tt
  where
  isSup : ⌈ A ⌉ → Bool
  isSup sup   = true
  isSup [ _ ] = false

\begin{code}
  _⌜∙⌝_ : ⌈ 𝑀 ⌉ → ⌈ 𝑀 ⌉ → ⌈ 𝑀 ⌉
  _      ⌜∙⌝ sup    = sup
  sup    ⌜∙⌝ [ _ ]  = sup
  [ x ]  ⌜∙⌝ [ y ]  = [ x ∙ y ]
